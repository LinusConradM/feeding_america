% Tighten paragraph spacing
\setlength{\parskip}{0.5em}   % space between paragraphs
\setlength{\parindent}{0em}   % remove paragraph indentation

% Tighten list spacing
\usepackage{enumitem}
\setlist{nosep}               % removes extra space in bullet and numbered lists

% Tighten heading spacing safely
\usepackage{sectsty}

\sectionfont{\vspace{-0.8em}}       % less space above section titles
\subsectionfont{\vspace{-0.6em}}    % less space above subsection titles
\subsubsectionfont{\vspace{-0.5em}} % less space above subsubsection titles
